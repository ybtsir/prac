<head> 
 <title>PRINT TABLE</title> 
 <script type="text/javascript"> 
  function multiTable_ForLoop(){ 
   var num = prompt("Enter number"); 
   for(var i=1; i<=10; i++){ 
    document.write(num + "x" + i + "=" + (num*i) + "<br>"); 
   } 
  } 
  function multiTable_WhileLoop(){ 
   var num = prompt("Enter number"); 
   var i=1; 
   while(i<=10){ 
    document.write(num + "x" + i + "=" + (num*i) + "<br>"); 
    i++; 
   } 
  } 
  function multiTable_DoWhileLoop(){ 
   var num = prompt("Enter number"); 
   var i=1; 
   do{ 
    document.write(num + "x" + i + "=" + (num*i) + "<br>"); 
    i++; 
   } 
   while(i<=10); 
  } 
</script>
