<!DOCTYPE HTML> 
<html> 
<head> 
 <title>PRINT TABLE</title> 
 <script type="text/javascript"> 
  function multiTable_ForLoop(){ 
   var num = prompt("Enter number"); 
   for(var i=1; i<=10; i++){ 
    document.write(num + "x" + i + "=" + (num*i) + "<br>"); 
   } 
  } 
  function multiTable_WhileLoop(){ 
   var num = prompt("Enter number"); 
   var i=1; 
   while(i<=10){ 
    document.write(num + "x" + i + "=" + (num*i) + "<br>"); 
    i++; 
   } 
  } 
  function multiTable_DoWhileLoop(){ 
   var num = prompt("Enter number"); 
   var i=1; 
   do{ 
    document.write(num + "x" + i + "=" + (num*i) + "<br>"); 
    i++; 
   } 
   while(i<=10); 
  } 
</script> 
</head> 
<body> 
<h1> Multiplication table</h1> 
<button onclick = "multiTable_ForLoop()"> using for_loop </button> 
<button onclick = "multiTable_WhileLoop()"> using while_loop </button> 
<button onclick = "multiTable_DoWhileLoop()"> using do_while_loop </button>
</body>
